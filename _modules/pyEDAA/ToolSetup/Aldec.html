@export
class Aldec(Vendor):
	_toolClasses: Dict[str, Tool] = {
		"Active-HDL": ActiveHDL,
		"Riviera-PRO": RivieraPRO,
	}

	@property
	def ActiveHDL(self) -> ActiveHDL:
		return self.__getitem__("Active-HDL")

	def RivieraPRO(self) -> RivieraPRO:
		return self.__getitem__("Riviera-PRO")
